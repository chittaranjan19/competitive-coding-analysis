ll mod = 1e9 + 7;

ll gcd(ll a, ll b) {
	return b ? gcd(b, a % b) : a;
}

ll _gcd(ll a, ll b, ll & x, ll & y) {
	if (a == 0) {
		x = 0; y = 1;
		return b;
	}
	ll _x1, _y1;
	ll d = _gcd(b%a, a, _x1, _y1);
	x = _y1 - (b / a) * _x1;
	y = _x1;
	return d;
}

ll binpow(ll a, ll n) {
	ll res = 1;
	while (n) {
		if (n & 1)
			res *= a;
		a *= a;
		n >>= 1;
		a %= mod;
		res %= mod;
	}
	return res;
}

ll n, m;

set<ll> r, c;

int main()
{
	cin >> n >> m;
	ll ans = n*n;
	while (m--){
		ll x, y;
		cin >> x >> y;
		if (!r.count(x))
			ans -= n - c.size();
		if (!c.count(y)) 
			ans -= n - r.size();
		if (!r.count(x) && !c.count(y))
			ans++;
		c.insert(y);
		r.insert(x);
		cout << ans << ' ';
	}
	return 0;
}
