int main() {
    // freopen("577B.in", "r", stdin);
    
    int n, m;
    scanf("%d %d", &n, &m);

    vector<bool> dp(m, false);
    while (n--) {
        int val;
        scanf("%d", &val);
        val %= m;
        for (int it = m-1; it >= 0; --it)
            if (dp[it])
                dp[(it+val)%m] = true;
        dp[val] = true;
    }
    if (dp[0] == true)
        puts("YES");
    else
        puts("NO");

    return 0;
}
