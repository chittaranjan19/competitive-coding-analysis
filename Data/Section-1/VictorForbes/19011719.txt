#include <bits/stdc++.h>

using namespace std;

#define N 500001
#define l first
#define r second

vector<pair<int, int>> v;
int dp[N];
int n;

int solve(int x){
	if (x >= n){
		return 0;
	}

	if (dp[x] != -1){
		return dp[x];
	}

	auto it = upper_bound(v.begin(), v.end(), make_pair(v[x].r, INT_MAX));

	return dp[x] = max(solve(it - v.begin()) + 1, solve(x + 1)); // Pegar ou nÃ£o pegar.
}

int main(){
	int i, x, y;

	scanf("%d", &n);

	for (i = 0; i < n; i++){
		scanf("%d%d", &x, &y);

		v.push_back({x, y});
	}

	sort(v.begin(), v.end());

	memset(dp, -1, sizeof(dp));

	printf("%d\n", solve(0));

	return 0;
}
