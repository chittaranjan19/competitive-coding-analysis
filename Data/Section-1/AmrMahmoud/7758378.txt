using namespace std;

int main()
{
    int n,m,a,b;
    cin >> n >> m >> a >> b;
    int temp = n/m;
    int res = 0;
    if(temp * b < m * temp * a)
    {
        res+=temp * b;
    }
    else res+=m * temp * a;
    res+=(n - m * temp) * a;
    cout << res;
    return 0;
}
