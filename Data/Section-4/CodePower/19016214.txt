using System;
using System.Linq;

namespace CF_mex
{
	class Program
	{
		static void Main(string[] args)
		{
			int n = Convert.ToInt32(Console.ReadLine());

			int[] arr = Console.ReadLine().Split(' ').Select(s => Convert.ToInt32(s)).OrderBy(x => x).ToArray();

			arr[0] = 1;
			for (int i = 1; i < n; i++)
				if (arr[i] > arr[i - 1])
					arr[i] = arr[i - 1] + 1;

			Console.Write(arr[n - 1] + 1);
		}
	}
}
