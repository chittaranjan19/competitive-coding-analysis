int main()
{
	tint n;
	cin >> n;
	vector<tint> a (n);
	forn(i,n)
		cin >> a[i];
	while (a[0] > 0)
	{
		forn(j,n)
		{
			if (j % 2 == 0)
				a[j] = ( a[j] + 1 ) % n;
			else
				a[j] = ( a[j] - 1 + n) % n;
		}
	}
	bool ans = true;
	forn(i,n)
		ans &= (a[i] == i);
	if (ans)
		cout << "Yes" << endl;
	else
		cout << "No" << endl;

}
