{

	tint n,m,k;
	cin >> n >> m >> k;
	vector<tint> p (n);
	forn(i,n)
		cin >> p[i];
	vector<tint> a (n-m+1);
	tint total = 0;
	forn(i,m)
		total += p[i];
	forsn(i,m,n+1)
	{
		a[i-m] = total;
		if (i < n)
		{
			total -= p[i-m];
			total += p[i];
		}
	}
	sort(a.begin(),a.end());
