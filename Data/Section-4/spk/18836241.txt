long long gcd(long long x,long long y)
{
    if(y==0)return x;
    return gcd(y,x%y);
}

long long lcm(long long x,long long y)   { return (x*y/gcd(x,y)); }

int main(void)
{
    int n,k,now=1;
    cin>>n>>k;

    while(n--)
    {
        int x;
        scan(x);

        now = gcd(k,lcm(now,x));
    }

    if(now==k)cout<<"Yes";
    else      cout<<"No";
    return 0;
}
