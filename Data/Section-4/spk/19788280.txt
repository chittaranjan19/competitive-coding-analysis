double cal(int a,int b,int c,int d)
{
    return sqrt((c-a)*(c-a) + (d-b)*(d-b));
}
int main(void)
{
  int a,b;
  cin>>a>>b;

  int n;
  cin>>n;

  double res=999999999999;
  while(n--)
  {
      int c,d;
      double v;
      cin>>c>>d>>v;

      double dist = cal(a,b,c,d);
      double time = dist/v;
      res = min(res,time);
  }
  printf("%.9f",res);
  return 0;
}
