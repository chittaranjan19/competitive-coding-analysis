int n, m, b;
ll mod;
ll a[510];
ll dp[510][510][510];


int main(){
    cin >> n >> m >> b >> mod;
    REPP(i, 1, n+1) cin >> a[i];
    CLEAR0(dp);
    dp[0][0][0] = 1;
    for(int i = 0; i<=m; i++){
        for(int j = 1; j<=n; j++){
            for(int k = 0; k<=b; k++){
                if(a[j] <= k && i) dp[i][j][k] = (dp[i][j][k] + dp[i-1][j][k-a[j]])%mod;
                dp[i][j][k] = (dp[i][j][k] + dp[i][j-1][k])%mod;
            }
        }
    }
    
    ll ans = 0LL;
    REP(i, b+1) ans = (ans+dp[m][n][i])%mod;
    cout << ans << endl;
}
