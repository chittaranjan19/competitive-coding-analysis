#include <iostream>

using std::cin;
using std::cout;
using std::endl;

int main(int argc, const char * argv[])
{
    
    int n,arr[26] = {0},i;
    char c;
    
    cin >> n;
    
    for(i=0;i<n;i++)
    {
        cin >> c;
        
        if(int(c) > 91)
            arr[c-97]++;
        else
            arr[c-65]++;
    }
    
    
    for(i=0;i<26;i++)
    {
        if(arr[i] == 0)
            break;
    }
    
    if(i==26)
        cout << "YES";
    else
        cout << "NO";
    
    cout << endl;

    
    return 0;
}
