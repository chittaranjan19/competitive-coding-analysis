#define MAX_N 16
int arr[MAX_N];

int main() {
#ifdef DEBUG
	freopen("550B.in", "r", stdin);
#endif
    
    int n, l, r, x;
    scanf("%d %d %d %d", &n, &l, &r, &x);
    rep(idx, n) {
    	scanf("%d", arr + idx);
    }
    int ans = 0;
    rep1(bm, (1 << n) - 1) {
    	int df_tot = 0;
    	int df_min = INF, df_max = 0;
    	rep(idx, n) {
    		if (bm & (1 << idx)) {
    			df_tot += arr[idx];
    			df_max = max(df_max, arr[idx]);
    			df_min = min(df_min, arr[idx]);
    		}
    	}
    	if ((l <= df_tot) and (df_tot <= r) and (x <= df_max - df_min)) {
    		ans++;
    	}
    }
    printf("%d\n", ans);

#ifdef DEBUG
    rep_rt();
#endif
    return 0;
}
