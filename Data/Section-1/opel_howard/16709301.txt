int main() {
    // freopen("632A.in", "r", stdin);
    
    int n, p;
    scanf("%d %d%*c", &n, &p);
    vs buys;
    while (n--) {
    	string str;
    	getline(cin, str);
    	buys.pub(str);
    }
    reverse(all(buys));

    ll ans = 0;
    ll n_apples = 0;
    iterate(str, buys) {
    	ans += n_apples*p;
    	n_apples <<= 1;
    	if (*str != "half") {
    		ans += (p >> 1);
    		n_apples++;
    	}
    }
    cout << ans << endl;

    return 0;
}
