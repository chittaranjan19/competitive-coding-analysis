using System;


class Program
{


    static void Main(string[] args)
    {
        string preoutput;
        string output = "";
        string str = Console.ReadLine();
        int a = Convert.ToInt32(str);
        if (a % 2 == 0)
            preoutput = "I love it";
        else
            preoutput = "I hate it";



        for (int i = 1; i < a; i++)
        {
            output += (i % 2 == 1) ? "I hate that " : "I love that ";
        }
        output += preoutput;

        Console.WriteLine(output);
    }
}
