typedef long long ll;
ll c(int n,int m)
{
    ll ans=1;
    int t=1;
    for(int i=n;i>=n-m+1;i--) {ans*=i,ans/=t,t++;}
    return ans;
}

{
    int n;
    while(cin>>n)

        ll ans=4*(n-1);
        for(int i=1;i<=n-2;i++) ans*=3;
        printf("%I64d\n",ans);
    }
}
