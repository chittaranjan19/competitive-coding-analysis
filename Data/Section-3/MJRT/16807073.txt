const int N = 5 + 10000;
bool vis[N];
int n;
int t[N];

int main()
{
    //ios::sync_with_stdio(false);
#ifndef ONLINE_JUDGE
    //freopen("in.txt", "r", stdin);

    while(1 == scanf("%d",&n))
    {
        memset(vis,0,sizeof(vis));
        int tot = 0;
        for(int i = 0 ; i < n ; i++)
        {
            int x;
            scanf("%d",&x);
            if(!vis[x])
            {
                t[tot++] = x;
                vis[x] = 1;
            }
        }
        sort(t,t+tot);
        bool flag = 0;
        for(int i = 0 ; i < tot-2 ; i ++)
        {
            if(t[i]+1 == t[i+1] && t[i+1]+1 == t[i+2])
            {
                flag = 1;
                break;
            }
        }

        puts(flag?"YES":"NO");
    }
    return 0;
}
