int main(void)
{
  int n,a;
  cin >> n >> a;

  int crim[n];

  for(int i=0;i<n;i++) cin >> crim[i];

  int r,l,ans=0;
  --a;
  if(crim[a]) ans++;
  for(r=a+1,l=a-1;r<n || l>=0;r++,l--)
  {
      if(r<n)
      {
          if(l>=0 && crim[l]==1 && crim[r]==1) ans+=2;
          else if(l<0 && crim[r]==1) ans++;
      }
      else
      {
          if(l>=0 && crim[l]) ans++;
      }
  }

  cout << ans;
  return 0;
}
