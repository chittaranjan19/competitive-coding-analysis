int main() {

    int k;
    scanf("%d", &k);
    string str;
    cin >> str;
    vi count(26, 0);
    loop(it, str.length()) {
    	int val = str[it]-'a';
    	count[val]++;
    }

    bool poss = true;
    string ans = "";
    loop(it, 26) {
    	if (count[it]%k)
    		poss = false;
    	ans += string(count[it]/k, it+'a');
    }
    if (!poss) {
    	puts("-1");
    	return 0;
    }
    loop(it, k)
    	cout << ans;
    cout << endl;

    return 0;
}
