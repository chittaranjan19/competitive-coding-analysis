#include <iostream>

int main() {
    /*
     Песня длится T секунд; прогружаются первые S секунд композиции
     и только потом включается песня; когда доходит до недогрузки,
     сразу включается заново; все это, пока песня полностью не прогрузится;
     за q секунд реального времени грузит (q-1) секунд(у) трека.
     Сколько раз он запустит песню?
     T, S, q (2 ≤ q ≤ 10^4, 1 ≤ S < T ≤ 10^5).
     
     например S = 4c; u == [6c за 7c]; =>
     => за 7с проигрывания готово 10с   ∆ = 3c
     14с <=> 16с    ∆ = 2
     21с <=> 22с    ∆ = 1
     28с <=> 28с    ∆ = 0
     
     7с <=> 34с     ∆ = 27
     14с <=> 40с    ∆ = 26
     ...
     [7*26 + 14; 6*26 + 40] =>
     196c = 196c    ∆ = 0c
     
     7c <=> 202c    ∆ = 195c
     ...
     1372c <=> 1372c
     
     1: ∆ = 4; p = 28;
     2: ∆ = 28; p = 196;
     3: ∆ = 196; p = 1372;
    */
    int t = 0, s = 0, q = 0;
    scanf("%d %d %d", &t, &s, &q);
    int raz = 0;
    if((q >= 2 && q <= 10000) && t <= 100000 && s >= 1 && t > s) {
        for (; s < t;raz++)
            s = s * q;
        printf("%d", raz);
    }
    else printf("%d", 0);
    return 0;
}
