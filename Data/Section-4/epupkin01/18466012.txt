int main()
{
    const ll k1 = 1234567, k2 = 123456, k3 = 1234;
    ll n;
    cin >> n;
    
    for (int i = 0; i * k1 <= n; ++i)
    {
        for (int j = 0; i * k1 + j * k2 <= n; ++j)
        {
            if ( (n - i * k1 - j * k2) % k3 == 0)
            {
                cout << "YES\n";
                return 0;
            }
        }
    }
    
    cout << "NO\n";
    
    return 0;
}
