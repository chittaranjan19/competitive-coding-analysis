#include <bits/stdc++.h>

using namespace std;

int main() {
    // m - mno≈ºnik, d - magiczna
    int m, d;
    scanf("%d %d", &m, &d);

    // (a,b) - zakres
    int a, b;
    scanf("%d %d", &a, &b);

    int i = a - a % m + m;
    long long wynik = 0;
    for (; i <= b; i += m) {
        ostringstream ss;
        ss << i;
        string str = ss.str();
        for (int j = 1; j < str.length(); j += 2)
            if (str[j] - '0' != d)
                goto eskejp;
        for (int j = 0; j < str.length(); j += 2)
            if (str[j] - '0' == d)
                goto eskejp;
        wynik++;
        wynik %= 1000000007;
        // printf("%d\n", i);
        eskejp: ;
    }
    
    printf("%lld", wynik);
}
