int main(void)
{
  double d,h,v,e;

  cin >> d >> h >> v >> e;

  double esp=0,r=(d/2);

  esp = v/(pie*r*r);
  esp = esp - e;

  if(esp<=0)
  {
      cout << "NO";
  }
  else
  {
      cout << "YES" << endl;
      double ans = h/(esp);
      printf("%lf",ans);
  }

  return 0;
}
