int main(void)
{
  int n;
  cin>>n;
  char st[n+1];
  scanf("%s",st);

  int last,curr,res=1000000000,flag=1;
  for(int i=0;i<n;i++)
  {
      if(i==0)scan(last);
      else
      {
          scan(curr);

          if(st[i-1]=='R' && st[i]=='L')
            res = min(res, abs(last-curr)/2),flag=0;

          last = curr;
      }
  }
  if(flag)         cout<<"-1";
  else             cout<<res;

  return 0;
}
