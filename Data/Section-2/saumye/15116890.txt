#include<stdio.h>
int main()
{
	int dp[501][501],i,j,h,w,q,r1,c1,r2,c2,ans;
	char mat[501][501];
	scanf("%d%d",&h,&w);
	for(i=0;i<h;i++)
		scanf("%s",mat[i]);
	for(j=0;j<=w;j++)
		dp[0][j]=0;
	for(i=1;i<=h;i++)
		dp[i][0]=0;
	dp[1][1]=0;
	for(j=2;j<=w;j++)
	{
		if(mat[0][j-1]=='.')
		{
			if(mat[0][j-2]=='.')
				dp[1][j]=dp[1][j-1]+1;
			else
				dp[1][j]=dp[1][j-1];
		}
		else
			dp[1][j]=dp[1][j-1];
	}
	for(i=2;i<=h;i++)
	{
		if(mat[i-1][0]=='.')
		{
			if(mat[i-2][0]=='.')
				dp[i][1]=dp[i-1][1]+1;
			else
				dp[i][1]=dp[i-1][1];
		}
		else
			dp[i][1]=dp[i-1][1];
	}
	for(i=2;i<=h;i++)
	{
		for(j=2;j<=w;j++)
		{
			dp[i][j]=dp[i-1][j]+dp[i][j-1]-dp[i-1][j-1];
			if(mat[i-1][j-1]=='.')
			{
				if(mat[i-2][j-1]=='.')
					dp[i][j]++;
				if(mat[i-1][j-2]=='.')
					dp[i][j]++;
			}
		}
	}
	/*for(i=0;i<=h;i++)
	{
		for(j=0;j<=w;j++)
			printf("%d ",dp[i][j]);
		printf("\n");
	}*/
	scanf("%d",&q);
	while(q--)
	{
		scanf("%d%d%d%d",&r1,&c1,&r2,&c2);
		ans=dp[r2][c2]-dp[r2][c1-1]-dp[r1-1][c2]+dp[r1-1][c1-1];
		if(r1>1)
		{
			for(j=c1;j<=c2;j++)
			{
				if(mat[r1-1][j-1]=='.' && mat[r1-2][j-1]=='.')
					ans--;
			}
		}
		if(c1>1)
		{
			for(i=r1;i<=r2;i++)
			{
				if(mat[i-1][c1-1]=='.' && mat[i-1][c1-2]=='.')
					ans--;
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
