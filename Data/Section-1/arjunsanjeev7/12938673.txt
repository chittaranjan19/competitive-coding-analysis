void FastIO()
{
	ios_base::sync_with_stdio(0);

VI V;

bool possible(int n)
{
	vector<LL> v;
	rep1(i,V.size())
	{
		int x=V[i];
		if(n%x==0)
		{
			v.pb(x);
		}
	}
	if(v.size()<2) return false;
	LL G=v[0];
	rep1(i,v.size())
	{
		G=(G*(LL)v[i])/(__gcd(v[i],G));
		if(G>n) return false;
	}
	if(G==n) return true;
	return false;
}

int main()
{
	FastIO();
	int n;
	cin>>n;
	int ans=n;
	rep2(i,1,n)
	{
		if(i==1 || (possible(i))) ans--;
		else V.pb(i);
	}
	cout<<ans<<endl;
	rep1(i,ans)
	{
		cout<<V[i]<<" ";
	}
	cout<<endl;
	return 0;
}
