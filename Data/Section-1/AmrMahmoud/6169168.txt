int main()
{
    int n;
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++)
        scanf("%d",&arr[i]);
    sort(arr,arr+n);
    LL res=0;
    for(int i=0;i<n;i++)
        res+=abs((i + 1) - arr[i]);
    printf("%I64d",res);
    return 0;
}
