    iter(e, arr)
    	cin >> *e;
    ll ans = INT_MAX;
    for (int idx = 0; idx < n-1; ++idx) {
    	if ((str[idx] == 'R') and (str[idx+1] == 'L')) {
    		ll l = arr[idx], r = arr[idx+1];
    		ll mid = (l+r) >> 1;
    		ans = min(ans, mid-l);
    	}
    }
    if (ans == INT_MAX) {
    	puts("-1");
    	return 0;
    }

    cout << ans << endl;
