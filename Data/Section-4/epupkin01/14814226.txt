#include <iostream>
#include <algorithm>

int main() {
    using namespace std;
    int n;
    cin >> n;
    int element[100002] = {0}, ne = 0;
    for (int i = 1, k; i <= n; i++) {
        cin >> k;
        element[k]++;
        ne = max(ne, k);
    }
    // main
    int ans = 0;
    for (int i = ne; i >= 0; i--) {
        element[i] = max(element[i + 1], element[i + 2] + element[i] * i);
        ans = max(element[i], ans);
    }
    cout << ans << endl;
    // print
    /*
    for (int i = ne; i > 0; i--) {
        cout << " " << element[i];
    }
    cout << endl;
    */
    return 0;
}
