ll gcd(ll a, ll b) {
	if (a < b) { ll tmp = a; a = b; b = tmp; }
	while (b != 0) { ll tmp = a; a = b; b = tmp % b; }
	return a;
}

ll lcm(ll a, ll b) {
	ll g = gcd(a, b); return g * a/g * b/g;
}

int main() {
	int n, k; cin >> n >> k;
	ll x = 1;
	for (int i = 0; i < n; i++) {
		int tmp; cin >> tmp;
		x = lcm(x, (ll)tmp);
		x = gcd(x, (ll)k);
	}
	if (x % k == 0) cout << "Yes" << endl;
	else cout << "No" << endl;

}
