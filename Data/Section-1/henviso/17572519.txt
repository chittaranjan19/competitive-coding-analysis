int main(){
		cin >> d >> h >> v >> e;
		double r = d/2.0;
		double ta = (10000.0 * e + h) * r * r  * PI;
		if(ta < v * 10000.0){
			cout << "YES\n";
			printf("%.10lf\n", (h * r * r * PI) / (v - e * r * r * PI) ); 
		}
		else cout << "NO\n";
}
