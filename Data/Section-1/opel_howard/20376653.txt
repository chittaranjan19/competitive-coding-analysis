int dp[MAXK + 1][MAXN + 1];

int main() {

    int n, k;
    scanf("%d %d", &n, &k);
    rep1(val, n) {
        dp[0][val] = 1;
    }
    rep(idx, (k - 1)) {
        rep1(val, n) {
            for (int j = 1; j * val <= n; ++j) {
                dp[idx + 1][j * val] = (dp[idx + 1][j * val] + dp[idx][val]) % MOD;
            }
        }
    }
    ll ans = 0;
    rep1(val, n) {
        ans = (ans + dp[k - 1][val]) % MOD;
    }
    printf("%I64d\n", ans);

    rep_rt();
#endif
    return 0;
}
