int main(){
	int p[100];
	int n, m, k, res = 0;
	int devices = 0;
	scanf(" %d  %d  %d ", &n, &m, &k);
	REP(i, n) scanf(" %d ", &p[i]);
	sort(p, p+n);
	for(int i = n-1; i>-1; i--){
		if(devices+k >= m) break;
		if(k){
			devices += p[i];
			res++;
			k--;
		}
		else{
			devices += (p[i]-1);
			res++;
		}
	}
	if(devices+k >= m) printf("%d\n", res);
	else printf("-1\n");
}
