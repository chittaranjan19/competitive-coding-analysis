int main(void)
{
  int n;
  char c;

  cin >> n;

  if(n>26)
  {
      cout << "-1";
      return 0;
  }

  int mp[26]={0};
  for(int i=0;i<n;i++)
  {
      cin >> c;
      mp[c-'a']++;
  }

  int ans=0;
  for(int i=0;i<26;i++)
  {
      if(mp[i])
        ans+=(mp[i]-1);
  }

  cout << ans;
  return 0;
}
