#include <cstdio>

int main() {
    int x = 0, ans = 0, d = 0, xx = 0;
    scanf("%d", &x);
    if (x == 1)
        ans = 1;
    while (x > 1) {
        d = 1;
        xx = x;
        while (xx) {
            xx = xx / 2;
            d = d * 2;
        }
        d = d / 2;
        x = x - d;
        ans = ans + 1;
        if (x == 1)
            ans++;
    }
    printf("%d\n", ans);
    return 0;
}
