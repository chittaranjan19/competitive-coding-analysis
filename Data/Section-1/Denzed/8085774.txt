bus = list(map(list, '''+------------------------+
|#.#.#.#.#.#.#.#.#.#.#.|D|)
|#.#.#.#.#.#.#.#.#.#.#.|.|
|#.......................|
|#.#.#.#.#.#.#.#.#.#.#.|.|)
+------------------------+'''.split('\n')))
places = [[1, 1], [2, 1], [3, 1], [4, 1]] + [[1 << y, 1 + 2 * x] for x in range(1,11) for y in range(3)]
for x, y in places[:int(input())]:
    bus[x][y] = 'O'
print(*map(lambda x: ''.join(x), bus), sep = '\n')
