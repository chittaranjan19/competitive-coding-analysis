int main(void)
{
  int a,b;
  long long time = 0;

  cin >> a >> b;

  while(a>0 && b>0)
  {
      if(a<2 && b<2)
         break;
      else if(a==2 || a==1 || a<=b)
      {
          a++;
          b-=2;
      }
      else
      {
          b++;
          a-=2;
      }
      time++;
  }

  cout << time;
  return 0;
}
