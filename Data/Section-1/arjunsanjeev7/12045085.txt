#define LL long long int
#define PII pair<int,int>
#define PLL pair<LL,LL>
#define VI vector<int>
#define SI set<int>
#define VPII vector< PII >
#define SPII set< PII >
#define mod (int)1e9+7
#define inf INT_MAX

using namespace std;

int main()
{
	LL n,m=-1,ans;
	cin>>n;
	map<int,int> M,N,O;
	LL a[n];
	VI num;
	rep1(i,n)
	{
		cin>>a[i];
		M[a[i]]++;
		if(!N[a[i]]) {N[a[i]]=i+1; O[a[i]]=i+1;}
		else O[a[i]]=i+1;
	}
	tr(M,it)
	{
		if(it->s > m)
		{
			m=it->s;
			num.clear();
			num.pb(it->f);
		}
		else if(it->s == m)
		{
			num.pb(it->f);
		}
	}
	m=inf;
	rep1(i,num.size())
	{
		int x=num[i];
		if(O[x]-N[x]<m)
		{
			m=O[x]-N[x];
			ans=x;
		}
	}
	cout<<N[ans]<<" "<<O[ans]<<endl;
	return 0;
}
