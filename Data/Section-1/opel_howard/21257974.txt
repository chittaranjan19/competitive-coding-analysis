int main() {
#ifdef DEBUG
    freopen("719B.in", "r", stdin);
#endif
    
    int n;
    scanf("%d%*c", &n);
    string str;
    getline(cin, str);
    char cc[2] = {'r', 'b'};
    int ac = 0, bc = 0;
    rep(idx, str.length()) {
        char c = str[idx];
        if (c == cc[idx % 2])
            continue;
        if (c == 'r')
            ac++;
        else
            bc++;
    }
    int ans = min(ac, bc) + abs(ac - bc);
    swap(cc[0], cc[1]);
    ac = 0, bc = 0;
    rep(idx, str.length()) {
        char c = str[idx];
        if (c == cc[idx % 2])
            continue;
        if (c == 'r')
            ac++;
        else
            bc++;
    }
    ans = min(ans, min(ac, bc) + abs(ac - bc));
    printf("%d\n", ans);

#ifdef DEBUG
    rep_rt();
#endif
    return 0;
}
