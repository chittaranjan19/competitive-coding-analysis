void FastIO()

int main()
{
	FastIO();
	int n,m,flag=0;
	set<int> S1,S2;
	cin>>n>>m;
	VI f(100001,0),b(100001,0),pos(100001,0);
	rep1(i,n)
	{
		cin>>f[i];
		pos[f[i]]=i;
		if(present1(S1,f[i]))
		{
			flag=1;
		}
		S1.insert(f[i]);
	}
	rep1(i,m)
	{
		cin>>b[i];
		S2.insert(b[i]);
	}
	tr(S2,it)
	{
		if(!present1(S1,*it))
		{
			cout<<"Impossible"<<endl;
			return 0;
		}
	}
	if(flag)
	{
		cout<<"Ambiguity"<<endl;
		return 0;
	}
	cout<<"Possible"<<endl;
	rep1(i,n)
	{
		cout<<1+pos[b[i]]<<" ";
	}
	cout<<endl;
	return 0;
}
