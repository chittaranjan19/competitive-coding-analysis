#include <iostream>
#include <cstring>
#include <algorithm>
#include <cstdlib>  // rand(), max(), min()
#include <ctime>    // rand(): srand(int(time(NULL)));
#define ll long long

using namespace std;

ll dv[61] = {1, 2, 4, 8, 16, 32, 64, 128, 256, 512, 1024, 2048, 4096, 8192, 16384, 32768, 65536, 131072, 262144, 524288, 1048576, 2097152, 4194304, 8388608, 16777216, 33554432, 67108864, 134217728, 268435456, 536870912, 1073741824, 2147483648, 4294967296, 8589934592, 17179869184, 34359738368, 68719476736, 137438953472, 274877906944, 549755813888, 1099511627776, 2199023255552, 4398046511104, 8796093022208, 17592186044416, 35184372088832, 70368744177664, 140737488355328, 281474976710656, 562949953421312, 1125899906842624, 2251799813685248, 4503599627370496, 9007199254740992, 18014398509481984, 36028797018963968, 72057594037927936, 144115188075855872, 288230376151711744, 576460752303423488, 1152921504606846976}, a, b;
int ans = 0;

void bod(int na, int nb) {
    ll delta;
    if (na == 1) {
        na = 2;
    }
    for (int i = na; i <= nb; i++) {
        delta = dv[i - 1] + dv[i - 2] - 1;
        if (delta >= a && delta <= b) {
            ans++;
        }
        for (ll j = i - 3; j >= 0; j--) {
            delta += dv[j];
            if (delta >= a && delta <= b) {
                ans++;
            }
        }
    }
}

int main() {
    cin >> a >> b;
    ll a1 = a, b1 = b;
    int na = 0, nb = 0;
    while (a1) {
        a1 /= 2;
        na++;
    }
    while (b1) {
        b1 /= 2;
        nb++;
    }
    bod(na, nb);
    cout << ans << endl;
    return 0;
}
