using namespace std;

bool DP[105][10][2];
string s;
string res;

bool can(int ind, int m, bool t)
{
  if(DP[ind][m][t]) return false;
  if(ind == s.size()) return (m == 0 && t);
  DP[ind][m][t] = true;
  if(can(ind + 1, m, t)) return true;
  int nxt = m * 10 + s[ind] - '0';
  nxt%=8;
  if(can(ind + 1, nxt, 1))
  {
    res+=s[ind];
    return true;
  }
  return false;
}

int main()
{
  ios_base::sync_with_stdio(0);
  cin >> s;
  if(can(0, 0, 0))
  {
    cout << "YES\n";
    reverse(res.begin(), res.end());
    cout << res;
  }
  else cout << "NO";
  return 0;
}
