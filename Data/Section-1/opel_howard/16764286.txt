vector<vector<int> > m;

int dp(int a, int b) {
	if ((a <= 0) or (b <= 0))
		return 0;
	if (m[a][b] != -1)
		return m[a][b];
	m[a][b] = max(dp(a+1, b-2), dp(a-2, b+1))+1;
	return m[a][b];
}

    scanf("%d %d", &a1, &a2);
    printf("%d\n", dp(a1, a2));

    return 0;
}
