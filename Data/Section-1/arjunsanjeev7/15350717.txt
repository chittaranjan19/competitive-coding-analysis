bool beautiful(string s)
{
	if(s[0]!='1') return false;
	for(int i=1;i<s.length();i++)
	{
		if(s[i]!='0') return false;
	}
	return true;
}

int main()
{
	FastIO();
	int n,cnt=0,flag=0;
	cin>>n;
	string s,t="1";
	for(int i=0;i<n;i++)
	{
		cin>>s;
		if(s=="0") flag=1;
		if(!beautiful(s))
		{
			t=s;
		}
		else
		{
			cnt+=(s.length()-1);
		}
	}
	if(flag)
	{
		cout<<"0"<<endl;
		return 0;
	}
	cout<<t;
	for(int i=0;i<cnt;i++)
	{
		cout<<"0";
	}
	cout<<endl;
	return 0;
}
