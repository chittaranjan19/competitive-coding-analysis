#define MAX_N (int) (1e5)

int arr[MAX_N + 1];

int main() {
#ifdef DEBUG
	freopen("545D.in", "r", stdin);
#endif
    
    int n;
    scanf("%d", &n);
    rep(idx, n) {
    	scanf("%d", &arr[idx]);
    }
    sort(rng(arr, n));

    int ans = 0;
    int pt = 0;
    int l = 0;
    while (l != n) {
    	int r = n;
    	while (l < r) {
    		int m = (l + r) >> 1;
    		if (arr[m] < pt) {
    			l = m + 1;
    		} else {
    			r = m;
    		}
    	}
    	if (l != n) {
    		ans++;
    		pt += arr[l++];
    	}
    }
    printf("%d\n", ans);

#ifdef DEBUG
    rep_rt();
#endif
    return 0;
}
