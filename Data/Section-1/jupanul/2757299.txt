#include<stdio.h>#include<algorithm>using namespace std;int month[]={0,31,28,31,30, 31, 30, 31, 31, 30, 31, 30, 31};int dbm  []={0, 0,31,59,89,120,150,181,212,242,273,303,334};struct MyStruct {int x,t,p;} v[205];inline MyStruct make(int a,int b,int c){    MyStruct v;    v.x=a;v.t=b;v.p=c;    return v;}class MyComp{public: inline bool operator()(const MyStruct &a,const MyStruct &b)    {        return a.x==b.x?(a.t<b.t):(a.x<b.x);    }};int main(){    freopen("input.txt","r",stdin);    freopen("output.txt","w",stdout);    int u=0,n,i,m,d,p,t;    scanf("%d",&n);    for(i=1;i<=n;i++)    {        scanf("%d%d%d%d",&m,&d,&p,&t);        int start,end;        end=dbm[m]+d-1;        start=end-t+1;        v[++u]=make(start,0,p);        v[++u]=make(end,1,p);    }    sort(v+1,v+u+1,MyComp());    int c,cmax;    c=cmax=0;    for(i=1;i<=u;i++)    {        if(v[i].t==0)            c=c+v[i].p;        else            c=c-v[i].p;        if(c>cmax)            cmax=c;    }    printf("%d\n",cmax);    return 0;}
