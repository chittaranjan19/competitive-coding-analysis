#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int main()
{
    int a, b, c;
    cin >> a >> b >> c;
    if(a == b) cout << "YES" << endl;
    else if(c == 0) cout << "NO" << endl;
    else if(1LL * (b - a) * c < 0) cout << "NO" << endl;
    else if((b - a) % c == 0) cout << "YES" << endl;
    else cout << "NO" << endl;
    return 0;
}
