using namespace std;

void scan(scantype &x); //IMPORTANT: for positive only

int main(void)
{
  int n,x,f=0,ans;
  cin >> n;

  long long sum=0;
  map<long long,int> mp;

  scan(x);
  mp[x] = 1;
  sum = x;
  ans = n-1;

  f = max(f,mp[x]);

  for(int i=1;i<n;i++)
   {
       scan(x);

       sum+= x;
       mp[sum]++;

       f = max(f,mp[sum]);
       ans = min(ans,n-f);
   }

  cout << ans;
  return 0;
}
