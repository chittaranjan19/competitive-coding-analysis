#define VAR(a,b) __typeof(b) a=(b)

int main()
{
	string s;
	cin >> s;
	int len=s.size();
	for(int i=122;i>=97;i--)
	{
		int flag=0;
		for(int j=0;j<len;j++)
		{
			if(s[j]==i)
			{
				flag=1;
				printf("%c",s[j]);
			}
		}
		if(flag==1)
			break;
	}
	cout  << endl;
	return 0;
}
