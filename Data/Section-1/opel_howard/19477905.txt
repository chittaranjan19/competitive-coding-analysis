#define N (int) (1e5)
#define MAX_VAL (int) (1e5)

int arr[N + 1];
bool c[MAX_VAL + 1];
int dp[N + 1];

int main() {
#ifdef DEBUG
	freopen("368B.in", "r", stdin);
#endif
    
    int n, m;
    scanf("%d %d", &n, &m);
    rep(idx, n) {
    	scanf("%d", &arr[idx]);
    }
    dp[n] = 0;
    for (int idx = n - 1; idx >= 0; --idx) {
    	dp[idx] = dp[idx + 1];
    	if (!c[arr[idx]]) {
    		c[arr[idx]] = true;
    		dp[idx]++;
    	}
    }
    while (m--) {
    	int l;
    	scanf("%d", &l);
    	printf("%d\n", dp[l - 1]);
    }

#ifdef DEBUG
    rep_rt();
#endif
    return 0;
}
