#include <bits/stdc++.h>
#include <iostream>

using namespace std;

int main()
{
    int n;
    cin >> n;
    int a[n];
    for (int i = 0; i < n; i++)
    {
        cin >> a[i];
        while (a[i] % 2 == 0)
            a[i] /= 2;
        while (a[i] % 3 == 0)
            a[i] /= 3;
    }
    for (int i = 0; i < n - 1; i++)
        if (a[i] != a[i + 1])
        {
            cout << "NO";
            return 0;
        }
    cout << "YES";
    return 0;
}
