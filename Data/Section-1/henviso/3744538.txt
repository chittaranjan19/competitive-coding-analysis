ull gcd(ull a, ull b){
	if(b == 0) return a;
	gcd(b, a%b);
}

int LOG10(ull a){
	int res = 0;
	while(a){
		a /= 10;
		res++;
	}
	return res;
}

ull lcm(ull a, ull b){
	ull aux = a/gcd(a,b);
	//cout << " A " << a << " GCD " << gcd(a,b) << endl;
	//cout << " LOG 1 " << LOG10(aux) << " LOG 2 " << LOG10(b) << endl;
	int casas = max(LOG10(aux), LOG10(b)); 
	if(LOG10(aux*b) < casas || LOG10(aux)+LOG10(b) > 19) return 0;
	return aux*b;
}
