int main() {
    // freopen("686A.in", "r", stdin);
    
    int n;
    ll x;
    scanf("%d %I64d%*c", &n, &x);

    int sad_kid = 0;
    while (n--) {
    	char op;
    	int d;
    	scanf("%c %d%*c", &op, &d);
    	if (op == '+')
    		x += d;
    	else {
    		if (d > x)
    			sad_kid++;
    		else
    			x -= d;
    	}
    }
    printf("%I64d %d\n", x, sad_kid);

    return 0;
}
