#define LL long long int
#define PII pair<int,int>
#define VI vector<int>
#define inf INT_MAX

using namespace std;

int main()
{
	int n,t,ans=0,add;
	cin>>n>>t;
	vector< vector< double > > arr(12,vector<double>(12,0));
	for(int i=0;i<t;i++)
	{
		add=1;
		for(int j=0;j<10;j++)
		{
			for(int k=0;k<=j;k++)
			{
				arr[j][k]+=add;
				if(arr[j][k]>1)
				{
					arr[j+1][k]+=((arr[j][k]-1)/2);
					arr[j+1][k+1]+=((arr[j][k]-1)/2);
					arr[j][k]=1;
				}
			}
			add=0;
		}
	}
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<=i;j++)
		{
			if(arr[i][j]>=1) ans++;
		}
	}
	cout<<ans<<endl;
	return 0;
}
