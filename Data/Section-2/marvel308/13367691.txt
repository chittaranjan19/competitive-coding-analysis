#include<stdio.h>
int main()
{
    int n,a[1005]={0},i,j,cnt=0,ans=0;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    while(cnt<n)
    {
        for(i=0;i<n;i++)
        {
            if(a[i]==-1)
                continue;
            if(cnt>=a[i])
            {
                a[i]=-1;
                cnt++;
            }
        }
        if(cnt==n)
            break;
        ans++;
        for(i=n-1;i>=0;i--)
        {
            if(a[i]==-1)
                continue;
            if(cnt>=a[i])
            {
                a[i]=-1;
                cnt++;
            }
        }
        if(cnt==n)
            break;
        ans++;
    }
    printf("%d\n",ans);
    return 0;
}
