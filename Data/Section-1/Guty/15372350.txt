int main()
{
	ios_base::sync_with_stdio(0);
	tint l,r,k;
	cin >> l >> r >> k;
	tint p = 1;
	tint ans = 0;
	while (p <= r)
	{
		if (p >= l)
		{
			cout << p << " ";
			ans++;
		}
		if (p > (r+k-1)/k)

		p *= k;
	}
