#include <cstdio>
#include <iostream>
#include <cstring>
#include <string>

#include <cctype>
#include <stack>

#include <cmath>
using namespace std;
typedef long long ll;
int main()
{
    char s[105];
    while(cin>>s)
    {
        int l=strlen(s);
        for(int i=0;i<l;i++)
        {
            s[i]=tolower(s[i]);
            if(s[i]=='a'||s[i]=='o'||s[i]=='y'||s[i]=='e'||s[i]=='u'||s[i]=='i') continue;
            printf(".%c",s[i]);
        }
        printf("\n");
    }
    return 0;
}
