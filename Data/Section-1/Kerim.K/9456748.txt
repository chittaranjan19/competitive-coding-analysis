int main()

	{
		int bl=0;
		b=a;
		b=b+c;

		b[j]=a[j];

		b[j+1]=a[j];
		for(int j=0;j<b.size()/2;j++)
		if(b[j]!=b[b.size()-1-j] and b[j]!='(')
		{
			b[b.size()-1-j]=b[j];
			bl++;
		}

			cout<<b<<endl;

		}
