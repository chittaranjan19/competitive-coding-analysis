int main(void)
{
  int sum=0;

  int freq[103]={0};

  for(int i=0;i<5;i++)
  {
      int x;
      cin >> x;
      freq[x]++;
      sum+=x;
  }

  int ans=sum;
  for(int i=100;i>0;i--)
  {
      if(freq[i]>1)
      {
          int t = min(freq[i],3);
          int temp = sum - (t * i);
          ans = min(ans,temp);
      }
  }

  cout << ans;
  return 0;
}
