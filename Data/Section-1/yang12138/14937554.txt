using namespace std;
typedef long long ll;
const int mod=1000000007;
int main()
{
    ll a,b;
    while(cin>>a>>b)
    {
        ll ans=((a+1)*b+2);
        ans%=mod;
        (ans*=a)%=mod;
        (ans*=b)%=mod;
        (ans*=(b-1))%=mod;
        ans/=4;
        cout<<ans<<endl;
    }
    return 0;
}
