int main() {
    // freopen("621A.in", "r", stdin);


    int n;
    scanf("%d", &n);
    vi arr(n);
    iterate(val, arr) {
    	scanf("%d", val);
    }
    sort(all(arr));

    ll tot = 0;
    ll odd = 0;
    bool is_even = true;
    for (int it = n-1; it >= 0; --it) {
    	ll val = arr[it];
    	if (val%2) {
    		odd += val;
    		is_even = !is_even;
    	} else {
    		tot += val;
    	}
    	if (is_even) {
    		tot += odd;
    		odd = 0;
    	}
    }
    printf("%I64d\n", tot);

    return 0;
}
