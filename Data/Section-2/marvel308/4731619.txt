#include<stdio.h>
#include<math.h>
int a[1000005]={0},b[1000005]={0};
char s[1000005];
int main()
{
scanf("%s",s);
int i,c=0,x=0,l=0,k=0,j;
for(j=0;s[j];j++)
{
if(s[j]=='M')
{
a[j]=1;
l++;
}
else
{
a[j]=0;
c++;
}
}
//printf("c=%d\n",c);
for(i=0;i<j;i++)
{
if(a[i]==1)
break;
x++;
}
if(l==0||x==c)
printf("0\n");
else
{x=0;
while(x!=c)
{
x=0;
k++;
for(i=0;i<j;i++)
{
if(a[i]==0&&i>0)
b[i]=a[i]|a[i-1];
if(a[i]==1&&i<j-1)
b[i]=a[i]&a[i+1];
}
for(i=0;i<j;i++)
{a[i]=b[i];
//printf("%d",a[i]);
}
//printf("\n");
for(i=0;i<j;i++)
{
if(a[i]==1)
break;
x++;
}
}
printf("%d\n",k);
}
return 0;
}
