#include <iostream>

struct Link
{
    long long int k1,k2;
};

Link box[100000+50];

int compare (const void * a, const void * b);

int main()
{

    cin>>N;
    
    cin>>x1>>x2;
    for(int j=0;j<N;j++)
    {
        int a,b;
        cin>>a>>b;
        long long int kk1=a*x1+b;
        long long int kk2=a*x2+b;
        
        box[j].k1=kk1;
        box[j].k2=kk2;
    }
    qsort(box,N,sizeof(Link),compare);

        cout<<"box[j]:"<<box[j].k1<<"~"<<box[j].k2<<endl;

        //cout<<"jã€j-1 :"<<j+1<<j<<endl;
        //cout<<"box[j-1]:"<<box[j-1].k1<<"~"<<box[j-1].k2<<endl;
        
        
        if(box[j].k1<box[j-1].k1&&box[j].k2>box[j-1].k2)
            ans=1;
    }
    /*for(int j=1;j<N;j++)

        {
            ans=1;
        }
    }*/
    if(ans==1)
        cout<<"YES"<<endl;

}

int compare (const void * a, const void * b)
{

  Link *orderA = (Link *)a;
  Link *orderB = (Link *)b;

  return ( orderB->k1 - orderA->k1 );
}
