using namespace std;
typedef long long ll;
ll c(int n,int x)
{
    double ans=1;
    double t=1;
    for(int i=n;i>=n-x+1;i--)
    {
        ans*=i;
        ans/=t;
        t++;
    }
    ll res=ll(ans);
    return res;
}
int main()
{
    int n;
    while(cin>>n)
    {
        printf("%I64d\n",c(n,5)+c(n,6)+c(n,7));
    }
    return 0;
}
