int n,a[200005],v[200005],t[200005],root,nr;

void getRoot(int nod){
    if(v[nod]){
        root=nod;
        a[nod]=nod;
        nr++;
    }
    else{
        v[nod]=1;
        getRoot(a[nod]);
    }
}

int nxt=1;

void DFS(int nod,int pas,bool init){
    if(v[nod]==pas && nod!=root){
        a[nod]=root;
        nr++;
        return;
    }
    if(v[nod]) return;
    v[nod]=pas;
    DFS(a[nod],pas,false);
    if(init){
        while(nxt<=n){
            if(!v[nxt]){
                DFS(nxt,pas+1,true);
                return;
            }
            nxt++;
        }
    }
}

int main(){
    cin >> n;
    for(int i=1;i<=n;i++) cin >> a[i];
    for(int i=1;i<=n;i++){
        if(!root && a[i]==i) root=a[i];
        else if(a[i]==i) a[i]=root, nr++;
    }
    if(!root) getRoot(1);
    for(int i=1;i<=n;i++) v[i]=0;
    DFS(root,1,true);
    cout << nr << '\n';
    for(int i=1;i<=n;i++) cout << a[i] << ' ';
}
