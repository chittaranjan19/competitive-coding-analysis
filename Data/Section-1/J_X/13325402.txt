#include <iostream>

using namespace std;

int main()
{
    int a, b;
    while(cin >> a >> b)
    {
        int ma = max(a, b), mi = min(a, b);
        cout << mi << ' ' << (ma - mi) / 2 << endl;
    }
    return 0;
}
