int main() {
    // freopen("672A.in", "r", stdin);
    
    int n;
    scanf("%d", &n);
    int n_digit = 1;
    int within_this_digit = 9;
    int start = 1;
    while (n > within_this_digit) {
    	n -= within_this_digit;
    	within_this_digit *= 10;
    	n_digit++;
    	start *= 10;
    }
    start += (n/n_digit-1);
    if (n%n_digit)
    	start++;
    string str = to_string(start);
    if (n%n_digit == 0)
    	cout << str.back() << endl;
    else
    	cout << str[(n%n_digit)-1] << endl;

    return 0;
}
