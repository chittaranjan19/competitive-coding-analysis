function main()
{
    var n,m,ans=0;
    redline(n);
    redline(m);
    while(m!=n)
    {
        if(m>n)
        {
            if(m%2!=0)
            {
                m++;
                ans++;
            }
            else
            {
                m/=2;
                ans++;
            }
        }
        else
        {
            ans+=n-m;
            m=n;
        }
    }
    print(ans);
}
main();
