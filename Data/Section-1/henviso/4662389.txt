int n;
ii v[100100];

int ans[100100][2];

int main(){
	while(cin >> n){
		REP(i, n){
			cin >> v[i].first;
			v[i].second = i+1;
		}
		sort(v, v+n);
		
		int qnt = 0;
		REP(i, n){
			int j = i;
			int r = 0;
			bool pos = true;
			
			while(j + 1 < n && v[j+1].first == v[i].first){
				j++;
				int nr = v[j].second-v[j-1].second;
				if(r){
					if(nr != r) pos = false;
				}
				else r = nr;
			}
			if(pos){
				ans[qnt][0] = v[i].first;
				ans[qnt][1] = r;
				qnt++;
			}
			i = j;
		}
		printf("%d\n", qnt);
		REP(i, qnt) printf("%d %d\n", ans[i][0], ans[i][1]);
	}
}
