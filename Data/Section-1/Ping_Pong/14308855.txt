#include <stdio.h>
#include <iostream>
#include <stdlib.h>
#include <algorithm>
#include <vector>
#include <set>
#include <queue>
#include <string>
#include <math.h>

using namespace std;

int main()
{
    int n;
    scanf("%d", &n);
    int a[n], b[n - 1], c[n - 2];
    for (int i = 0; i < n; i++)
        scanf("%d", &a[i]);
    for (int i = 0; i < n - 1; i++)
        scanf("%d", &b[i]);
    for (int i = 0; i < n - 2; i++)
        scanf("%d", &c[i]);
    sort(a, a + n);
    sort(b, b + n - 1);
    sort(c, c + n - 2);
    for (int i = 0; i < n; i++)
        if (i == n - 1)
            printf("%d\n", a[i]);
        else
            if (a[i] != b[i])
            {
                printf("%d\n", a[i]);
                break;
            }
    for (int i = 0; i < n - 1; i++)
        if (i == n - 2)
            printf("%d\n", b[i]);
        else
            if (b[i] != c[i])
            {
                printf("%d\n", b[i]);
                break;
            }
    return 0;
}
