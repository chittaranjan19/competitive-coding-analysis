int n,a[200005],v[200005],root,nr;
vector <int> g[200005];


int nxt=1;
void DFS(int nod, bool init){
    if(v[nod]) return;
    v[nod]=1;
    for(int i=0;i<g[nod].size();i++){
        //cout << nod << "->" << g[nod][i] << '\n';
        DFS(g[nod][i],false);
    }
    if(init){
        while(nxt<=n){
            if(!v[nxt]){
                a[nxt]=root;
                nr++;
                DFS(nxt,true);
                break;
            }
            nxt++;
        }
    }
}

int main(){
    cin >> n;
    for(int i=1;i<=n;i++) cin >> a[i];
    for(int i=1;i<=n;i++){
        v[i]=1;
        if(a[i]==i){
            if(root) a[i]=root, nr++;
            else root=i;
        }
    }
    if(!root) root=1, a[1]=1, nr++;
    for(int i=1;i<=n;i++) v[i]=0;
    for(int i=1;i<=n;i++) g[a[i]].push_back(i);
    DFS(root,true);
    cout << nr << '\n';
    for(int i=1;i<=n;i++) cout << a[i] << ' ';
}
