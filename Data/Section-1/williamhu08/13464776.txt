using namespace std; typedef pair<int, int> pii; const int MAXN = 2010; int N; int A[MAXN], pos[MAXN]; vector<pii> ans; void swpos (int x, int y) {swap(pos[A[x]], pos[A[y]]); swap(A[x], A[y]); ans.push_back(pii(x, y)); } int main() {if (fopen("input.txt", "r")) {freopen("input.txt", "r", stdin); } scanf("%d", &N); for (int i = 1; i <= N; i++) {scanf("%d", &A[i]); } for (int i = 1, x; i <= N; i++) {scanf("%d", &x); pos[x] = i; } int nmove = 0; for (int i = 1; i <= N; i++) {A[i] = pos[A[i]]; nmove += abs(A[i] - i); } printf("%d\n", nmove / 2);	/* div2 */ /* redefine pos */ for (int i = 1; i <= N; i++) {pos[A[i]] = i; } for (int i = N; i >= 1; i--) {/* move i to correct place... */ while (pos[i] != i) {for (int j = pos[i] + 1; j <= i; j++) {if (A[j] <= pos[i]) {/* whoops <= not < */ swpos(j, pos[i]); break; } } } } printf("%d\n", int(ans.size())); for (pii p : ans) {printf("%d %d\n", p.first, p.second); } }
