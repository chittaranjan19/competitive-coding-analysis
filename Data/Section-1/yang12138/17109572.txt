const int maxn=15;
int main()
{
    int n,m,t;cin>>n>>m>>t;
    char s[maxn][maxn];
    int ans=0;
    for(int i=0;i<n;i++){
        scanf("%s",s[i]);
        int now=0;
        for(int j=0;s[i][j];j++){
            if(s[i][j]=='Y') now++;
        }
        if(now>=t) ans++;
    }
