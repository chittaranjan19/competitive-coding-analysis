int n;
int a[1000000];
int p[1000000];

int main(){
    cin >> n;
    REP(i, n) cin >> a[i] >> p[i];
    int ans = 0;
    int sum = 0;
    REP(i, n){
        int j = i+1;
        sum = a[i];
        while(j<n && p[j] > p[i]){
            sum += a[j];
            j++;
        }
        ans += sum*p[i];
        i = j-1;
    }
    cout << ans << endl;
}
